#CHROMIUM
#/home/alarm/.xinitrc

#!/usr/bin/env sh

xset -dpms
xset s off
xset s noblank

unclutter &
chromium http://localhost:3000 \
  --kiosk \
  --start-fullscreen \
  --window-size=1080,1920 \
  --window-position=0,0 \
  --incognito \
  --noerrdialogs \
  --disable-translate \
  --no-first-run \
  --fast \
  --fast-start \
  --disable-infobars \
  --disable-features=TranslateUi \
  --disk-cache-dir=/dev/null \
  --overscroll-history-navigation=0 \
  --disable-pinch \
  --disable-features=AudioServiceOutOfProcess
